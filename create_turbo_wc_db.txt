CREATE TABLE bathrooms(
NAME varchar(256) PRIMARY KEY
);

CREATE TABLE stamps(
ID int(16) AUTO_INCREMENT PRIMARY KEY,
DATE DATETIME,
BATHROOM_NAME varchar(256),
FOREIGN KEY (BATHROOM_NAME) REFERENCES bathrooms(NAME)
);
